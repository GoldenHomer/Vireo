401  protected Attachment renamePrimaryDocument() {
402 
403  // We only enforce the nameing convention on primary documents.
404  if (this.type != AttachmentType.PRIMARY)
405  throw new IllegalStateException("Unable to rename the primary document on an attachment which is not of type = PRIMARY.");
406 
407  // Step 1) Figure out what the primary document should be named.
408  String namePart = null;
409  if (submission.getStudentLastName() != null && submission.getStudentLastName().trim().length() > 0)
410  // If available use the last name.
411  namePart = submission.getStudentLastName();
412  if (namePart == null && submission.getStudentLastName() != null && submission.getStudentLastName().trim().length() > 0)
413  // If the last name is unavailable, use the first name.
414  namePart = submission.getStudentFirstName();
415  if (namePart == null)
416  // Lastly fall back to the word primary if we don't have a student name;
417  namePart = "primary";
418 
419  String docPart = null;
420  if (submission.getDocumentType() != null && submission.getDocumentType().trim().length() > 0)
421  // Use the document type
422  docPart = submission.getDocumentType();
423  if (docPart == null)
424  // If no document type then, just use the word "document"
425  docPart = "document";
426 
427  String yearPart = null;
428  if (submission.getGraduationYear() != null)
429  // Add the year if helpfull
430  yearPart = String.valueOf(submission.getGraduationYear());
431 
432  // Put it all together
433  String filename = namePart + "-" + docPart;
434  if (yearPart != null)
435  filename += "-" + yearPart;
436 
437  // Sanitize the filename for any random characters
438  filename = Normalizer.normalize(filename, Normalizer.Form.NFD);
439  filename = filename.replaceAll("[^a-zA-Z0-9\\-_]", "");
440  filename = filename.toUpperCase();
441

443  if (extension != null && extension.trim().length() > 0)
444  filename += "." + extension;
445

447  Attachment exists = submission.findAttachmentByName(filename);
448  if (exists != null && exists != this) {
