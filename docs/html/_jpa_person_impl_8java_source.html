581  protected String generateHash(String message) {
582  try {
583  if (this.getId() == null)
584  throw new IllegalStateException("Unable to use a password on an unpersisted person object. You *must* call save before attepmting to set or verify a password.");
585 
586  // Get the password salt
587  byte[] salt = new byte[24];
588  Random random = new Random(Long.valueOf(this.getId()));
589  random.nextBytes(salt);
590 
591 
592  // Generate the hash using the pre-defined algorithm.
593  MessageDigest md = MessageDigest.getInstance(HASH_ALGORITHM);
594  md.update(salt);
595  byte[] byteHash = md.digest(message.getBytes());
596 
597  // Convert the hash to hex values for easy storage and portability.
598  StringBuffer hash = new StringBuffer();
599  for (int i = 0; i < byteHash.length; i++) {
600  hash.append(Integer.toString((byteHash[i] & 0xff) + 0x100, 16).substring(1));
601  }
602 
603  // Retun the hash.
604  return hash.toString();
605  } catch (NoSuchAlgorithmException nsae) {
606  throw new IllegalStateException(
607  "Unable to generate password hash because no such algorithm exists for: "
608  + HASH_ALGORITHM, nsae);
609  }
610 
611  }
