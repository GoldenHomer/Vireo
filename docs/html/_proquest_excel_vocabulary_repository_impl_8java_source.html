302 
303  // Read Workbook
304  Workbook wb;
305  InputStream is = new FileInputStream(file);
306  try {
307  String extension = FilenameUtils.getExtension(file.getName());
308  if ("xlsx".equals(extension)) {
309  // Newer XML office format
310  // TODO: This appears to cause memory problems? I guess for now use the old format? Perhaps in a newer version of apache-poi this will work better.
311  wb = new XSSFWorkbook(is);
312  } else {
313  // Older office 97 binary format
314  wb = new HSSFWorkbook(is);
315  }
316 
317  // Read the data
318  List<List<String>> rowsOut = new ArrayList<List<String>>();
319 
320  Sheet sheet = wb.getSheetAt(0);
321  Iterator<Row> rows = sheet.rowIterator();
322  while (rows.hasNext()) {
323  Row row = rows.next();
324 
325  List<String> cellsOut = new ArrayList<String>();
326  Iterator<Cell> cells = row.cellIterator();
327  while (cells.hasNext()) {
328  Cell cell = cells.next();
329  String value = cell.getRichStringCellValue().getString();
330  cellsOut.add(value.trim());
331  }
332 
333  rowsOut.add(cellsOut);
334  }
335 
336  return rowsOut;
337 
338  } finally {
339  wb = null;
340  is.close();
341  is = null;
342  System.gc();
343  }
