613  protected <T> List<T> decodeList(String encoded, Class<T> type) {
614 
615  String[] split = encoded.split(",");
616 
617  List<T> result = new ArrayList<T>();
618  for (String raw : split) {
619  if (raw.length() ==0)
620  continue;
621 
622  try {
623  if ( type == String.class) {
624  // List type is string
625  result.add( (T) unescape(raw));
626 
627  } else if (type == Integer.class) {
628  // List type is integer, grad month or year
629  Integer value = Integer.valueOf(raw);
630  result.add((T) value);
631 
632  } else if (type == Person.class){
633  // List type is person, just for assignee
634 
635  if ("null".equals(raw)) {
636  result.add(null); // unassigned
637  } else {
638  Long personId = Long.valueOf(raw);
639  Person person = personRepo.findPerson(personId);
640  result.add((T) person);
641  }
642  } else if (type == Submission.class) {
643  // List type is submission
644 
645  Long subId = Long.valueOf(raw);
646  Submission sub = subRepo.findSubmission(subId);
647  result.add((T) sub);
648 
649  } else if (type == ActionLog.class) {
650  // List type is ActionLog
651 
652  Long logId = Long.valueOf(raw);
653  ActionLog log = subRepo.findActionLog(logId);
654  result.add((T) log);
655 
656  } else if (type == EmbargoType.class) {
657  // List type is embargo types
658 
659  Long embargoId = Long.valueOf(raw);
660  EmbargoType embargo = settingRepo.findEmbargoType(embargoId);
661  result.add((T) embargo);
662 
663  } else if (type == Semester.class) {
664  // List type is graduation semestens: year/month
665  String[] semesterSplit = raw.split("/");
666 
667  Semester semester = new Semester();
668  if (!"null".equals(semesterSplit[0]))
669  semester.year = Integer.valueOf(semesterSplit[0]);
670  if (!"null".equals(semesterSplit[1]))
671  semester.month = Integer.valueOf(semesterSplit[1]);
672 
673  result.add((T) semester);
674 
675  } else if (type == SearchOrder.class) {
676  // List type is SearchOrder column
677  int columnId = Integer.valueOf(raw);
678  SearchOrder column = SearchOrder.find(columnId);
679  result.add((T) column);
680  } else if (type == CustomActionDefinition.class) {
681  // List type is CustomActionDefinition
682  Long customActionId = Long.valueOf(raw);
683  CustomActionDefinition customAction = settingRepo.findCustomActionDefinition(customActionId);
684  result.add((T) customAction);
685  } else {
686  throw new IllegalArgumentException("Unable to decode unexpected object type: "+ type);
687  }
688  } catch (RuntimeException re) {
689  // Just log the error but keep on trucking. One legitimate
690  // reason why this may fail is if a person has been deleted
691  // since this filter was created, thus the person's id would no
692  // longer be valid.
693  Logger.warn(re,"Unable to decode value '"+raw+"' for type "+type.getName());
694  }
695  }
696 
697  return result;
698  }

715  protected void encodeList(StringBuilder result, List<?> values) {
716 
717  boolean first = true;
718  for (Object value: values) {
719  if (first)
720  first = false;
721  else
722  result.append(",");
723 
724  if (value == null) {
725  result.append("null");
726  } else if (value instanceof String) {
727  // Plain old strings, the most common case.
728  result.append(escape((String)value));
729 
730  } else if (value instanceof Integer){
731  // Integers from grad month & year
732  result.append(String.valueOf((Integer) value));
733 
734  } else if (value instanceof Person) {
735  // Full person object from assignee
736  Long personId = ((Person) value).getId();
737  result.append(String.valueOf(personId));
738 
739  } else if (value instanceof Submission) {
740  // Full submission object
741  Long subId = ((Submission) value).getId();
742  result.append(String.valueOf(subId));
743 
744  } else if (value instanceof ActionLog) {
745  // Full actionlog object
746  Long logId = ((ActionLog) value).getId();
747  result.append(String.valueOf(logId));
748 
749  } else if (value instanceof EmbargoType) {
750  // Full embargo object
751  Long embargoId = ((EmbargoType) value).getId();
752  result.append(String.valueOf(embargoId));
753 
754  } else if (value instanceof SearchOrder) {
755  // Full SearchOrder(column) object
756  int columnId = ((SearchOrder)value).getId();
757  result.append(String.valueOf(columnId));
758 
759  } else if (value instanceof Semester) {
760  // Graduation semester: year/month
761 
762  Semester semester = (Semester) value;
763  if (semester.year == null)
764  result.append("null");
765  else
766  result.append(String.valueOf(semester.year));
767 
768  result.append("/");
769 
770  if (semester.month == null)
771  result.append("null");
772  else

774 
775  } else if (value instanceof CustomActionDefinition) {
776  // Full CustomActionDefinition object
777  Long customActionId = ((CustomActionDefinition)value).getId();
778  result.append(String.valueOf(customActionId));
779  } else {
780  throw new IllegalArgumentException("Unable to encode unexpected object type: "+value.getClass().getName());
781  }
782  }

785  result.append(":");
