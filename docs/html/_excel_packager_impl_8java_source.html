482  public ExportPackage generatePackage(Submission submission) {

484  throw new IllegalArgumentException("Unable to generate package because not attachment types have been defined.");
485  }
486 
487  // Check that we have everything that we need.
488  if (submission == null || submission.getId() == null)
489  throw new IllegalArgumentException("Unable to generate a package because the submission is null, or has not been persisted.");
490 
491  try {
492 
493  // Set String replacement parameters
494  Map<String, String> parameters = new HashMap<String, String>();
495  parameters = StringVariableReplacement.setParameters(submission);
496 
497  File pkg = null;
498  List<Attachment> actionLogAttachments = new ArrayList<Attachment>();
499

501 
502  // The package has more than one file, so export as a directory.
503  pkg.delete();
504  pkg.mkdir();
505

507  if (xl.exists() || xl.isDirectory()) {
508  xl.delete();
509  xl.createNewFile();

511  XSSFWorkbook wbook = writeWorkbook(submission, Arrays.asList(SearchOrder.values()));
512  FileOutputStream os = new FileOutputStream(xl);
513  wbook.write(os);
514  os.close();
515 
516  // Add all the attachments
517  List<Attachment> attachments = submission.getAttachments();
518  List<Attachment> actionLogs = submission.getAttachmentsByType(AttachmentType.ACTIONLOG);
519  // add them to this list to delete the temp files later
520  actionLogAttachments.addAll(actionLogs);
521  // add them to this list to add them to export
522  attachments.addAll(actionLogs);
523  for (Attachment attachment : attachments) {

526  continue;

531  String shortFileName = attachment.getName().replaceAll("." + FilenameUtils.getExtension(attachment.getName()), "");
532 
533  String fileName = attachment.getName();
534 
535  // Customize Attachment Name
536  if (attachmentAttributes.get(attachment.getType().name()).get("customName") != null) {

540  }
541
