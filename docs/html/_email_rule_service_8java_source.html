116  private static boolean ruleIsValid(EmailWorkflowRule ewflRule, Submission submission) {
117  boolean conditionValid = true, recipientsValid = true, ruleEnabled = true;
118  // check condition != null
119  if (ewflRule.getCondition() != null) {
120  // check condition type and id != null
121  if (ewflRule.getCondition().getConditionType() == null) {
122  conditionValid = false;
123  }
124  switch (ewflRule.getCondition().getConditionType()) {
125  case College:
126  case Department:
127  case Program:
128  if (ewflRule.getCondition().getConditionId() == null) {
129  conditionValid = false;
130  }
131  break;
132  case Always:
133  break;
134  default:
135  throw new UnsupportedOperationException();
136  }
137  } else {
138  conditionValid = false;
139  }
140  // check recipients
141  if (ewflRule.getRecipients(submission).size() == 0) {
142  recipientsValid = false;
143  }
144  // check to see if rule is disabled
145  ruleEnabled = (!ewflRule.isDisabled()); // if rule isDisabled(true) return false(rule is not enabled)
146  return (conditionValid && recipientsValid && ruleEnabled);

237  // always send to student on submission

239  studentRule.setCondition(conditionStudent); // always save a new copy of condition
240  studentRule.setEmailTemplate(studentEmailTemplate);
241  studentRule.setRecipientType(RecipientType.Student);
242  studentRule.setIsSystem(true);

261  }
262  }
263  }
264  }

268  advisorRule.getCondition().save();

270  advisorRule.save();
271  }
